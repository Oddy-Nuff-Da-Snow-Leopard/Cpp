#include <iostream>
#include <conio.h>
using namespace std;

{
	setlocale(LC_CTYPE, "Russian");

	cin >> a;
	if (a > 0)
		b = 1;
	else
		b = 0;
	while (a!= 000)
	{
		if (a > 0 & b == 0)
		{
			c++;
			b = 1;
		}
		else if (a < 0 & b == 1)

			c++;
			b = 0;
		}

}
