	int k_plus_f = k + f, k_plus_t = k + t, f_plus_t = f + t;

//7. Даны числа a, b, c. Если a < b < c, то каждое число заменить наибольшим,

#include <iostream>
#include <conio.h>
using namespace std;
void main()
{
	setlocale(LC_CTYPE, "Russian");
	int a, b, c;
	cout << "Введите числа : " << endl;
	cout << "a = "; cin >> a;
	cout << "b = "; cin >> b;
	cout << "c = "; cin >> c;
	cout << endl;
	if (a < b && b < c)
	{
		a = c; b = c;
		cout << "a = " << a << endl << "b = " << b << endl << "c = " << c << endl;
	}
	else if (a > b && b > c)
		cout << "a = " << a << endl << "b = " << b << endl << "c = " << c << endl;
	else
	{
		a *= a; b *= b; c *= c;
		cout << "a = " << a << endl << "b = " << b << endl << "c = " << c << endl;
	}
	
	_getch();
}

//12. Есть действительные числа x, y, z. Вычислить min(x+y+z / 2, xyz)+max(xy, y).
#include <iostream>
#include <conio.h>
#include <ctime>
using namespace std;
void main()
{
	setlocale(LC_CTYPE, "Russian");
	srand((unsigned)time(NULL));
	double a[3];
	int i = 0;
	for (i = 0; i < 3; i++)
		a[i] = rand() % 9 + 1;
	cout << "x = " << a[0] << "   " << "y = " << a[1] << "   " << "z = " << a[2] << endl;
	double x_plus_y_plus_z = 0, product_of_number_xyz = 1;
	for (i = 0; i < 3; i++)
	{
		x_plus_y_plus_z += a[i];
		product_of_number_xyz *= a[i];
	}
	x_plus_y_plus_z /= 2;
	double min;
	if (x_plus_y_plus_z < product_of_number_xyz)
		min = x_plus_y_plus_z;
	else
		min = product_of_number_xyz;
	cout << "min(x + y + z / 2, xyz)  : " << min << endl;
	double product_of_numbers_xy = a[0] * a[1];
	double max;
	if (product_of_numbers_xy > a[1])
		max = product_of_numbers_xy;
	else
		max = a[1];
	cout << "max(xy, y) : " << max << endl;
	cout << "min(x + y + z / 2, xyz) + max(xy, y) : " << min + max << endl;
	_getch();
}
