	cin.getline(s1, N);
	for (i = 0; s1[i] != '\0'; i++)
	{
		if (s1[i] == 'a')
		{
			s2[i + l] = s1[i];
			l++;
			s2[l + i] = '!';

		}
		else
			s2[i + l] = s1[i];
		r++;
	}
	s2[r+l] = '\0'; 
	for (i = 0; s2[i] != '\0'; i++)
		cout << s2[i];

}

10. Написать программу, которая записывает строку в обратном порядке.
#include <conio.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "");
	system("color f0");
	int i, m = 0, l = 0;
	const int N = 100;
	char s1[N], s2[N];
	cout << "Введите строку : " << endl;
	cin.getline(s1, N);
	for (i = 0; s1[i] != '\0'; i++)
	{
		m++;
	}
	m -= 1;
	for (i = 0, m; i < m+1, m >= 0; i++, m--)
	{
		s2[i] = s1[m];
		l++;
	}

	cout << endl;
	cout << "Перевернутая строка : " << endl;
	for (i = 0; i<l; i++)
	{
		cout << s2[i] ;
	}
	_getch();
}
