#include <iostream>
#include <iomanip>
#include <ctime>
#include <conio.h>
using namespace std;
void main()
{
	setlocale(0, "");
	system("color f0");
	int i, j, n, sum = 0, r = 0, l;
	const int N = 100;
	int matrix[N][N];
	srand((unsigned)time(NULL));
	cout << "Введите количество строк и столбцов : ";
	cin >> n;
	l = n;
	cout << endl;
	cout << "Ваша матрица размером " << n << " на " << n << endl << endl;
	for (i = 0; i < n; i++)
		for (j = 0; j < n; j++)
			matrix[i][j] = rand() % 4;
	for (i = 0; i < n; i++)
	{
		for (j = 0; j < n; j++)
		{
			cout << setw(4) << matrix[i][j];
		}
		cout << endl << endl;
	}
	cout << endl << endl;

	if (l % 2 == 0)
	{
		for (i = 0; i < n; i++)
		{
			for (j = r; j < n; j++)
			{
				sum += matrix[i][j];
			}
			n--;
			r++;
		}
		r--;
		for (i = n; i < l; i++)
		{
			for (j = r; j < n + 1; j++)
			{
				sum += matrix[i][j];
			}
			n++;
			r--;
		}
		cout << "Сумма : " << sum << endl;
		_getch();
	}
	else
	{
		for (i = 0; i < n; i++)
		{
			for (j = r; j < n; j++)
			{
				sum += matrix[i][j];
			}
			n--;
			r++;
		}
		r-=2;
		n++;
		for (i = n; i < l; i++)
		{
			for (j = r; j < n + 1; j++)
			{
				sum += matrix[i][j];
			}
			n++;
			r--;
		}
		cout << "Сумма : " << sum << endl;
		_getch();
	}
}
